Лабораторна робота №2.6

Dockerfile


              FROM mongo:latest
              ENV MONGO_INITDB_ROOT_USERNAME=admin
              ENV MONGO_INITDB_ROOT_PASSWORD=password
              ENV MONGO_INITDB_DATABASE=testdb
              EXPOSE 27017

docker-compose.yml


              version: '3.8'
              services: # Секція с сервисами
                mongodb:
                  image: mongo:latest # беру образ MongoDB
                  container_name: mongodb_container
                  restart: always
                  environment: # налаштунки MongoDB
                    MONGO_INITDB_ROOT_USERNAME: admin
                    MONGO_INITDB_ROOT_PASSWORD: password
                  ports:
                    - "27017:27017"
                  volumes: # підключаю том для храніння данних
                    - mongodb_data:/data/db
              volumes: # обьявляю тома
                mongodb_data:
