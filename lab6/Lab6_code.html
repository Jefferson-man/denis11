$(document).ready(function() {
                // тут фото по центру в верхній частині
                $('#photo').css({
                    'position': 'absolute',
                    'top': '20px',
                    'left': '50%',
                    'transform': 'translateX(-50%)'
                });
            
                // зробив відображення підпису під фото
                $('#caption').fadeIn(1000);
            
                // зменшую фото в 2 рази
                $('#photo').animate({
                    width: '75px',
                    height: '100px'
                }, 1000, function() {
                    // зробив рух по вертикалі до нижньої межі
                    $(this).animate({
                        top: $(window).height() - $(this).height() - 20
                    }, 2000, function() {
                        // 5. Заміна фото
                        $(this).attr('src', '/image/image6_2.jpg');
            
                        // прочрачність зроблюю 
                        $(this).fadeOut(1000);
                    });
                });
            });
