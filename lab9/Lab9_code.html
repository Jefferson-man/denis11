DTD схема


        <!DOCTYPE doctorsSchedule [
    <!ELEMENT doctorsSchedule (doctor+)>
    <!ELEMENT doctor (specialization, fullName, workingDays, time, office)>
    
    <!ELEMENT specialization (#PCDATA)>
    
    <!ELEMENT fullName (lastName, firstName, middleName)>
    <!ELEMENT lastName (#PCDATA)>
    <!ELEMENT firstName (#PCDATA)>
    <!ELEMENT middleName (#PCDATA)>
    
    <!ELEMENT workingDays (#PCDATA)>
    <!ELEMENT time (#PCDATA)>
    <!ELEMENT office (#PCDATA)>
    
    <!ATTLIST doctor doctorID ID #REQUIRED>
    <!ATTLIST office officeNumber ID #REQUIRED>
        ]>
